public class ReverseNumber {

    public static int reverseNumber(int n) {
        int reversed = 0;

        while (n != 0) {
            int digit = n % 10;           
            reversed = reversed * 10 + digit; 
            n = n / 10;                  
        }

        return reversed;
    }

    public static void main(String[] args) {
        int num = 1200;
        int result = reverseNumber(num);
        System.out.println("Reversed: " + result); 
    }
}
